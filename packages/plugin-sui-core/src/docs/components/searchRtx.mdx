## SearchRTX 搜索rtx名称

```js
import { SearchRTX } from 'sui-components';
```

### 何时使用

- 当搜索RTX名称的时候可以使用

### 使用方法

<CodeBox
  demo={[
    {
      component: (
        <SearchRTX style={{width: '200px'}} placeholder="请输入创建人"/>
      ),
      code: `import SearchRTX from 'sui-components';
// 可传antd中Select其他的api
// 不使用form表单时, 使用value、onChange的方式受控组件
<SearchRTX
  style={{width: '200px'}}
  placeholder='请输入创建人'
  value={this.state.creator}
  onChange={value => this.setState({
    creator: value
  })}
/>
// 使用form表单时
<Form.Item style={{width: '200px'}}>
  {getFieldDecorator('creator')(
    <SearchRTX
      placeholder='请输入创建人'
      restrictWrite={false}
    />
  )}
</Form.Item>;`
    }
  ]}
  api={[
    {
      title: '',
      data: [
        {
          key: '1',

        {

          params: 'onChange',

        },

          type: 'bool',

    }
